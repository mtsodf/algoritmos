  \begin{frame}[fragile]{Insertion Sort}

  \begin{frame}[fragile]{Merge Sort}

  \end{frame}                                                                                                                                            
                                                                                                                                                         
  \begin{frame}[fragile]{Quick Sort}                                                                                                                     
  \begin{tikzpicture}[scale=1]                                                                                                                           
      \node[draw] at (0,0) {5};                                                                                                                          
      \node[draw] at (1,0) {1};                                                                                                                          
      \node[draw] at (2,0) {4};                                                                                                                          
                                                                                                                                                         
      \node[draw] at (5,0) {1};                                                                                                                          
      \node[draw] at (6,0) {5};                                                                                                                          
      \node[draw] at (7,0) {4};                                                                                                                          
  \end{tikzpicture}                                                                                                                                      
                                                                                                                                                         
  Quick sort: \verb|[5,1,4]| $\rightarrow$ \verb|[1,5,4]|                                                                                                
  \end{frame}
